import requests
import json
url = "https://leaddistro.textdrip.com/api/create_lead/"
payload = json.dumps({
    "name": "Lead Name",
    "phone": "1234567890",
    "email": "",
    "birthdate": "",
    "address": "",
    "state": "",
    "zipcode": "",
    "custom_field1": "",
    "custom_field2": "",
    "custom_field3": "",
    "custom_field4": "",
    "custom_field5": "",
    "optional" : "usha"
})
headers = {
    'Content-Type': 'application/json',
    'APIKey': '{{profile.api_key}}'
}
response = requests.request("POST", url, headers=headers, data=payload)
print(response.text)
